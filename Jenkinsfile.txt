pipeline {
    agent any

    tools { nodejs 'Node23' }

    stages {
        stage('Install dependencies') {
            steps {
                sh 'npm ci --prefer-offline'
            }
        }
        stage('Running production backtests') {
            steps {
                sh 'npm install'
            }
        }
        stage('Running Automation') {
            steps {
                sh 'npx cypress run' or 'npm run cy:run:e2e'
            }
        }
    }
    post {
        always {
            junit 'results/*.xml'
        }
    }
}
